SNIPPET_ID: rag_index_builder
TAG: rag, retrieval, vectorstore
DESCRIPTION:
Скрипт для построения RAG‑индекса из текстов или документации.
Он разбивает тексты на чанки, генерирует embeddings и сохраняет индекс.
CODE:
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import FAISS
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.schema import Document

def build_rag_index(docs, persist_dir="./rag_index"):
    \"\"\"Построить векторный индекс на основе списка строк docs\"\"\"
    embeddings = OpenAIEmbeddings()
    splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=200)
    doc_objs = []
    for text in docs:
        chunks = splitter.split_text(text)
        for chunk in chunks:
            doc_objs.append(Document(page_content=chunk))
    vect = FAISS.from_documents(doc_objs, embeddings)
    vect.save_local(persist_dir)
    return vect

def rag_search(query: str, top_k: int=5) -> str:
    \"\"\"Поиск топ‑k релевантных контекстов по запросу query\"\"\"
    try:
        retriever = load_rag_retriever()
    except Exception as e:
        return \"RAG index not built yet.\"
    docs = retriever.get_relevant_documents(query)
    output = []
    for d in docs[:top_k]:
        output.append(d.page_content.strip())
    return \"\\n---\\n\".join(output)
